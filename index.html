        <script>
          window._chatData = window._chatData || [];
          function chatag(){_chatData.push(arguments);}
          chatag('js', new Date());
          chatag('config', 'aHR0cHM6Ly9kb2NzLnczY3ViLmNvbQ==');
        </script>
